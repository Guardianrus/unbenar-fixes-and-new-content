edict_protect_trade = {
	potential = {
		NOT = {
			FROM = {
				all_province_in_state = {
					OR = {
						has_building = town_hall
						has_building = government_lvl_3
						has_building = government_lvl_4
					}
				}
			}
		}
	}

	allow = {
		if = {
			limit = { ROOT = { ai = yes } }
			ROOT = {
				is_at_war = no
				has_country_modifier = xorme_ai_valid_nation
			}
		}
		else = {
			custom_trigger_tooltip = {
				tooltip = disable_state_edicts_active.tooltip
				ROOT = {
					NOT = {
						has_country_flag = disable_state_edicts_active
					}
				}
				always = yes
			}
		}
	}

	modifier = {
		province_trade_power_modifier = 0.5
		trade_value = 0.5
		local_state_maintenance_modifier = 0.25
	}

	color = {  113 11 43 }
	ai_will_do = {
		factor = 10.0
		modifier = {
			factor = 0.5
			any_province_in_state = {
				is_capital = yes
			}
		}
		modifier = {
			factor = 1.5
			all_province_in_state = {
				island = yes
			}
		}
		modifier = {
			factor = 1.5
			any_province_in_state = {
				province_trade_power =15
			}
		}
		modifier = {
			factor = 1.5
			any_province_in_state = {
				province_trade_power =25
			}
		}
		modifier = {
			factor = 1.5
			any_province_in_state = {
				province_trade_power =35
			}
		}
		modifier = {
			factor = 1.5
			any_province_in_state = {
				has_river_estuary_trigger =yes
			}
		}
		modifier = {
			factor = 1.5
			any_province_in_state = {
				OR = {
					province_has_center_of_trade_of_level = 1
					province_has_center_of_trade_of_level = 2
					province_has_center_of_trade_of_level = 3
				}
			}
		}
	}
}
