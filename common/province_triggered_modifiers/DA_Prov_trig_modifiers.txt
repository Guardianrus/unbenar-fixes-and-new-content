
DA_pr_tr_mithril_guard = {
	# Whether the triggered modifier is visible in the Triggered Modifiers window
	potential = {
		has_great_project = {
			type = The_Dwarven_Workshop
			tier = 1
		}
	}
	# Whether the triggered modifier is active
	trigger = {
		owner = {
			culture_group = dwarven
		}
		has_great_project = {
			type = The_Dwarven_Workshop
			tier = 2
		}
	}
    
	# The modifier to apply when the triggered modifier is active
	has_samurai = yes
	# samurai_cost_modifier = 1
	# Province scope
	on_activation = {
		owner = {
			add_country_modifier = {
				name = DA_can_recruit_Mithril_guard
				duration = -1
			}
		}
		add_permanent_province_modifier = {
			name = DA_can_recruit_Mithril_guard_pr
			duration = -1
		}
	}
	
	on_deactivation = {
		owner = {
			remove_country_modifier = DA_can_recruit_Mithril_guard
		}
		remove_province_modifier = DA_can_recruit_Mithril_guard_pr
	} 
}
