add_stability_or_adm_power = {
	if = {
		limit = {
			stability = 3
		}
		add_adm_power = 50
	}
	if = {
		limit = {
			NOT = { stability = 3 }
		}
		add_stability = 1
	}
}

reduce_stability_or_adm_power = {
	if = {
		limit = {
			stability = -2
		}
		add_stability = -1
	}
	else = { add_adm_power = -50 }
}
