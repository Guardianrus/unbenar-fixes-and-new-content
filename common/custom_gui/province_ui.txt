#centaur
custom_button = {
	name = province_icon_centaur_so
	potential = { has_province_modifier = centaur_minority_oppressed_small }
	trigger = { }
	effect = {
		tooltip = { add_province_modifier = { name = centaur_minority_oppressed_small_ui duration = -1 } }
		hidden_effect = {
			FROM = { 
				if = {
					limit = {
						NOT = { has_country_flag = racial_military_menu_opened }
						NOT = { has_country_flag = racial_pop_menu_opened }
						NOT = { has_country_modifier = racial_pop_menu_cooldown }
					}
					set_country_flag = racial_pop_menu_opened
					pop_menu_cored_dev_calculations = { RACE = centaur }
					country_event = { id = racial_pop_events_centaur.12 }
				}
			}
		}
	}
}
