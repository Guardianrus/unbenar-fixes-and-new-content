	any_neighbor_province = {
		owned_by = root
		NOT = {
			OR = {
				has_terrain = dwarven_hold
				has_terrain = dwarven_hold_surface
			}
		}
		NOT = { has_province_modifier = DA_secondary_Hold }
	}
}
