xorme_ai_great_power = {
	potential = {
		has_country_modifier = xorme_ai_valid_nation
		OR = {
			is_great_power = yes
			OR = {
				AND = {
					NOT = { is_year = 1500 }
					total_development = 175
				}
				AND = {
					is_year = 1500
					total_development = 525
				}	
			}
		}
	}

	trigger = {
		always = yes
	}
	army_tradition_decay = -0.01
	global_tax_income = 12
	state_maintenance_modifier = -0.35
}
