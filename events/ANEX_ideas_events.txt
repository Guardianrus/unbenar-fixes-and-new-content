province_event = {
	id = ANEX_ideas_events.1
	title = ANEX_ideas_events.1.t
	desc = ANEX_setting_event.1.d
	picture = WESTERNISATION_eventPicture

	hidden = yes

	trigger = {
		owner = {
			has_idea = zentra3
		}
		any_province_in_state = {
			is_capital = yes
		}
		NOT = {
			has_province_modifier = zentra_dev_1
			has_province_modifier = zentra_dev_0

		}
	}

	immediate = {
		log = "start ANEX_ideas_events.1 [GetDate] for province [Root.GetName]"
		if = {
			limit = {
				is_capital = yes
			}
			add_province_modifier = {
				name = zentra_dev_0
				duration = -1
				hidden = yes
			}
		}
		else = {
			add_province_modifier = {
				name = zentra_dev_1
				duration = -1
				hidden = yes
			}
		}
		log = "end ANEX_ideas_events.1 [GetDate] for province [Root.GetName]"
	}

	option = {
		name = ANEX_setting_event.1.a
		ai_chance = {
			factor = 1
		}
	}
}

province_event = {
	id = ANEX_ideas_events.28
	title = ANEX_setting_event.1.t
	desc = ANEX_setting_event.1.d
	picture = WESTERNISATION_eventPicture

	hidden = yes

	trigger = {
		OR = {
			AND = {
				owner = {
					has_idea = cultural_ties
				}
				AND = {
					NOT = {
						has_province_modifier = cultural_ties0
					}
					AND = {
						has_owner_accepted_culture = no
						has_owner_culture = no
						is_owned_by_trade_company = no
					}
				}
			}
			AND= {
				has_province_modifier = cultural_ties0
				OR = {
					owner = { NOT = { has_idea = cultural_ties } }
					has_owner_accepted_culture = yes
					has_owner_culture = yes
					is_owned_by_trade_company = yes
				}
			}
	
		}
	}

	immediate = {
		log = "start ANEX_ideas_events.28 [GetDate] for country [Root.GetName]"
		
		if = {
			limit = {
				owner = {
					has_idea = cultural_ties
				}
				AND = {
					NOT = {
						has_province_modifier = cultural_ties0
					}
					AND = {
						has_owner_accepted_culture = no
						has_owner_culture = no
						is_owned_by_trade_company = no
					}
				}
			}					
				
			add_province_modifier = {
				name = cultural_ties0
				duration = -1
				hidden = yes
			}
		}
		if = {
			limit = {
				OR = {
					AND = {
						has_province_modifier = cultural_ties0
						OR = {
							has_owner_accepted_culture = yes
							has_owner_culture = yes
							is_owned_by_trade_company = yes
						}
					}
					owner = {
						NOT = {
							has_idea = cultural_ties
						}
					}
				}
			}
			remove_province_modifier = cultural_ties0
		}
	}

	after = {
		log = "end ANEX_ideas_events.28 [GetDate] for country [Root.GetName]"
	}

	option = {
		name = ANEX_setting_event.1.a
		ai_chance = {
			factor = 1
		}
	}
}
